  1. ОБЩИЕ ПРАВИЛА
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  Роль в JWT: "STUDENT".

  Успешные ответы — JSON (кроме скачивания .ics — текст iCal).

  Типичная ошибка:
    { "statusCode": number, "message": string | string[], "path": string }

  Коды: 400 валидация, 401 нет/просрочен токен, 403 чужая роль или доступ,
        404 не найдено, 409 конфликт (например, уже откликался).

  2. МОДУЛИ БЭКЕНДА, ЗАТРАГИВАЮЩИЕ СТУДЕНТА (ЛОГИКА)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  auth              Регистрация, вход, refresh, logout
  users             Профиль студента (GET/PATCH /users/me)
  catalog           Города, категории, теги, комбинированный каталог для форм
  jobs              Лента вакансий, одна вакансия, рекомендации AI (matchScore)
  applications      Отклик, список своих откликов
  schedule          Источники расписания (PDF), слоты занятости
  upload            Presigned URL для загрузки файлов в S3/MinIO
  resume            Черновики резюме + AI-советы
  ai                Health, cover letter (POST), interview prep (GET)
  chat              HTTP: история и отправка сообщений по отклику
  notifications     Список уведомлений, «прочитано»
  job-alerts        Подписки на новые вакансии по фильтру
  profiles          Публичный профиль студента (без JWT)
  employer-reviews  Отзыв о работодателе; публичный список отзывов
  gamification      Очки и лидерборд
  internship        Трекер стажировки (часы, задачи — для студента/работодателя)
  skill-tests       Список тестов, прохождение, бейджи, история попыток
  video             Видеоинтервью Daily.co: присоединиться к комнате, созданной работодателем
  calendar          Экспорт расписания в .ics
  payments          Не для студента (Kaspi — работодатель Premium)
  health, swagger   Технические (health публичный по /api/health)

  3. AUTH  (/auth) — без JWT
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  POST /auth/register
    Тело:
      email       string (email)
      password    string (мин. 8 символов)
      role        "STUDENT"
      firstName   string, опционально
      lastName    string, опционально
    Ответ 201:
      accessToken   string
      refreshToken  string
      expiresIn     number (секунды жизни access)

  POST /auth/login
    Тело: { "email": string, "password": string }
    Ответ 200: как register (токены).

  POST /auth/refresh
    Тело: { "refreshToken": string }
    Ответ 200: новая пара токенов (как register).

  POST /auth/logout
    Тело: { "refreshToken": string }
    Ответ 204 без тела.

  4. USERS  (/users) — JWT
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  GET /users/me
    Ответ 200:
      id, email, role, createdAt
      profile (для STUDENT):
        firstName, lastName, phone, university, specialty, bio, portfolioUrl,
        timezone, avatarStorageKey,
        githubUsername (string | null),
        telegramChatId — маска "***linked***" или null (реальный chat_id не отдаётся)

  PATCH /users/me
    Тело (все поля опционально):
      firstName, lastName, phone, university, specialty, bio,
      portfolioUrl (валидный URL),
      timezone,
      avatarStorageKey — строка ключа S3 после /upload/presign (prefix uploads/{userId}/), или null,
      githubUsername — string | null
    Telegram: только через POST /telegram/link-token (не PATCH /users/me).
    Ответ 200: как GET /users/me.

  5. CATALOG  — без JWT (корневые пути без префикса /catalog кроме одного)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  GET /cities
    Ответ 200: массив городов
      id, name, slug, imageStorageKey, imageUrl, sortOrder, isActive, nameI18n { ru, kk, en }, даты

  GET /job-categories
    Ответ 200: массив категорий (parentId для дерева на фронте); imageUrl для превью

  GET /tags
    Ответ 200: массив тегов

  GET /catalog/job-form
    Ответ 200: { "cities": [...], "jobCategories": [...], "tags": [...] }
      Удобно одним запросом для экрана фильтров/поиска.

  6. JOBS  (/jobs)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  GET /jobs
    Авторизация: опциональна (Optional JWT).
    Query (все опционально):
      q, location, cityId, categoryId, tagId — строки/UUID фильтрации
      compatibleWithSchedule=true — только вакансии, совместимые с расписанием;
        требуется JWT с ролью STUDENT, иначе 400
    Ответ 200: массив опубликованных вакансий (status=PUBLISHED)
      Поля включают: id, title, description, status, cityId, location,
      salaryMin, salaryMax, currency, requiredWeeklyHours, workWindows,
      isPremium, source, externalId (для импорта HH),
      relations: city, categories[], tags[]
      Сортировка: премиум выше, затем по дате создания.

  GET /jobs/recommended
    Авторизация: JWT + роль STUDENT.
    Query:
      limit — число 1..50, по умолчанию 10
      compatibleWithSchedule — true/false, опционально
    Ответ 200: массив вакансий с полем matchScore (0–100) по сходству профиля
      (векторные эмбеддинги + опционально фильтр по расписанию).

  GET /jobs/:id
    Авторизация: опциональна.
    Ответ 200: одна вакансия с city, categories, tags.
      Для пользователя с ролью STUDENT дополнительно matchScore (0–100).
    404 если вакансия не опубликована и пользователь не владелец-работодатель.

  Создание/редактирование/удаление вакансий — только EMPLOYER (студенту не доступно).


━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  7. APPLICATIONS  (/applications) — JWT
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  POST /applications
    Роль: STUDENT.
    Тело:
      jobId       UUID (обязательно)
      coverLetter string, опционально (до ~8000 символов по DTO)
    Ответ 201: объект отклика
      id, jobId, studentUserId, studentProfileId, status, coverLetter,
      employerScore (nullable, заполняется фоновым AI при наличии OpenAI),
      createdAt, updatedAt
    Ошибки: 404 вакансия не PUBLISHED; 409 уже откликался; 403 нет профиля студента.

  GET /applications/me
    Роль: STUDENT.
    Ответ 200: массив откликов (сортировка createdAt DESC). Каждый элемент:
      id, jobId, studentUserId, studentProfileId, status, coverLetter,
      employerScore (0–100 или null), hasReviewed (boolean),
      createdAt, updatedAt,
      job: { id, title, location, salaryMin, salaryMax, currency, city },
      student: { id, email },
      studentProfile: { firstName, lastName, university, specialty }

  GET /applications/:id
    Роль: STUDENT (только свой отклик) или EMPLOYER (отклик на свою вакансию).
    Ответ 200: тот же контракт, что и элемент списка выше.
    hasReviewed — только для STUDENT.

  PATCH /applications/:id/withdraw
    Роль: STUDENT; только свой отклик.
    Тело: пустое.
    Ответ 200: отклик со status WITHDRAWN.
    Разрешено из: SUBMITTED, REVIEWING, SHORTLISTED, INTERVIEW.
    Запрещено из: OFFER, HIRED, REJECTED, WITHDRAWN; при ACTIVE internship — 400.
    Побочный эффект: уведомление работодателю.

  8. SCHEDULE  (/schedule) — JWT + STUDENT
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  POST /schedule/sources
    Тело:
      storageKey — ключ файла в S3 после POST /upload/presign и PUT файла
      mimeType   — MIME тип файла
    Ответ 201: ScheduleSource
      id, studentProfileId, storageKey, mimeType, parseStatus, createdAt, ...

  GET /schedule/sources
    Ответ 200: массив ScheduleSource студента.

  GET /schedule/slots
    Ответ 200: массив ScheduleSlot
      id, studentProfileId, dayOfWeek (0–6), startMinute, endMinute, label, ...

  PATCH /schedule/slots/:id
    Тело (частично): dayOfWeek, startMinute, endMinute, label
    Ответ 200: обновлённый ScheduleSlot.

  Парсинг PDF в фоне (BullMQ): после загрузки источника слоты могут появиться
  автоматически; статус смотреть в ScheduleSource.parseStatus.

  9. UPLOAD  (/upload) — JWT (любая роль)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  POST /upload/presign
    Тело:
      filename    string
      contentType string (MIME)
    Ответ 200:
      uploadUrl   string — PUT сюда файл с тем же Content-Type
      storageKey  string — передать в schedule/sources, users/me (avatar),
                 resume/drafts (pdf), и т.д.
      expiresIn   number (секунды)

  GET /media/url?storageKey=...
    JWT обязателен. storageKey должен начинаться с uploads/{yourUserId}/.
    Ответ 200: { url, expiresIn }.
    Публичные аватары: avatarUrl в GET /profiles/:userId (без raw storageKey).

  10. RESUME  (/resume) — JWT + STUDENT
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  GET /resume/drafts
    Ответ 200: массив черновиков ResumeDraft
      id, studentProfileId, title, contentJson, pdfStorageKey, createdAt, updatedAt

  POST /resume/drafts
    Тело:
      contentJson   object (обязательно) — схема на стороне фронта
      title         string, опционально
      pdfStorageKey string, опционально — после presign
    Ответ 201: ResumeDraft

  PATCH /resume/drafts/:id
    Тело: title, contentJson, pdfStorageKey (null — отвязать PDF)
    Ответ 200: ResumeDraft

  DELETE /resume/drafts/:id
    Ответ: успешное удаление (обычно 200/204 в зависимости от реализации сервиса)

  GET /resume/drafts/:id/suggestions
    Query: language — опционально ru | kk | en
    Ответ 200: объект с советами от AI (структура зависит от StudentAiService;
      обычно список рекомендаций/текст).

  11. AI  (/ai)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  GET /ai/health
    Без JWT.
    Ответ 200: { module, openaiEmbeddings: { configured, model } }

  POST /ai/cover-letter
    JWT + STUDENT.
    Тело:
      jobId   UUID (обязательно)
      language опционально: ru | kk | en
      tone    опционально: formal | friendly
    Ответ 200: { "text": string } — готовое сопроводительное письмо

  GET /ai/interview-prep
    JWT (любая авторизованная роль, не только студент).
    Query:
      jobId   UUID (обязательно)
      language опционально ru | kk | en
      count   опционально 3–20 — число вопросов
    Ответ 200: { "questions": string[] } (или аналогичная структура массива вопросов)

  12. CHAT  (/chat) — JWT
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  GET /chat/applications/:applicationId/messages
    Ответ 200: массив сообщений (ограниченная история, порядок по времени)
      id, roomId/applicationId, senderId, body, createdAt, ...

  POST /chat/applications/:applicationId/messages
    Тело: { "body": string } (1..8000 символов)
    Ответ 201: созданное сообщение

  WebSocket namespace /chat (не REST): события join, send — см. frontend-api-guide.txt.
  Handshake: JWT в auth.token или query token.

  20. VIDEO  (/video) — JWT
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  Сценарий для студента:
    1. Работодатель переводит заявку в INTERVIEW и открывает комнату (POST).
    2. Студент получает уведомление APPLICATION_UPDATE { videoRoom: true }.
    3. Студент вызывает GET — получает ссылку со своим participant-токеном.
    4. Открывает ссылку (новая вкладка).

  GET /video/rooms/:applicationId      — студент или работодатель (участник заявки)
    Доступно при статусе заявки: INTERVIEW, OFFER, HIRED (иначе 403).
    Если комната не создана — 404:
      { message: "Video room not created yet — waiting for employer to start" }
      Показать текст: «Ожидайте — работодатель откроет созвон».
    Ответ 200: { name, url, token, expiresAt }
      url уже содержит ?t=TOKEN — открывать как есть (новая вкладка).
    Без DAILY_API_KEY — demo URL (заглушка).

  UI-логика для СТУДЕНТА (карточка заявки):
    status = INTERVIEW и GET → 200 → кнопка «Присоединиться к интервью»
    status = INTERVIEW и GET → 404 → текст «Ожидайте — работодатель откроет созвон»
    Уведомление { videoRoom: true } → toast + активировать кнопку (повторить GET)
    status не из [INTERVIEW, OFFER, HIRED] → кнопка скрыта

  Примечание:
    Студент НЕ может создать комнату (POST доступен только EMPLOYER).
    POST от студента → 403 "Only employers can initiate video calls".
